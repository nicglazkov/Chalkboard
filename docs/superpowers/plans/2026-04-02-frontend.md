# Minimal Frontend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single-page web UI that lets a user generate a Chalkboard video without touching the CLI: enter a topic, watch the pipeline stages tick off in real-time, and watch or download the final video.

**Architecture:** A single `server/static/index.html` page served by FastAPI's `StaticFiles` mount (already wired in `server/app.py`). Vanilla HTML/CSS/JS — no build step, no npm. The page POSTs to `POST /api/jobs`, then connects to `GET /api/jobs/{id}/events` via the browser's native `EventSource` API for live progress. When the job completes, it shows a `<video>` player pointing at `GET /api/jobs/{id}/files/final.mp4` and download buttons for the SRT, quiz JSON, and script.

**Tech Stack:** HTML5, CSS3, vanilla JS (EventSource, fetch), served by FastAPI StaticFiles

**Prerequisites:** Plans 2 (pre-frontend cleanup) and 3 (FastAPI server) must be complete.

  <script>
    const STAGE_LABELS = {
      init: 'Initializing',
      research_agent: 'Researching topic',
      script_agent: 'Writing script',
      fact_validator: 'Fact-checking',
      manim_agent: 'Generating animation code',
      code_validator: 'Validating code',
      render_trigger: 'Synthesizing voiceover',
      render: 'Rendering animation',
      qa: 'Visual QA',
    };

    const form = document.getElementById('job-form');
    const submitBtn = document.getElementById('submit-btn');
    const statusEl = document.getElementById('status');
    const resultEl = document.getElementById('result');
    const errorEl = document.getElementById('error-msg');

    let currentJobId = null;
    let stageEls = {};

    function addStage(name, state = 'pending') {
      if (stageEls[name]) return;
      const div = document.createElement('div');
      div.className = 'stage';
      div.innerHTML = `<div class="dot ${state}" id="dot-${name}"></div>
                       <span>${STAGE_LABELS[name] || name}</span>`;
      statusEl.appendChild(div);
      stageEls[name] = div;
    }

    function setStage(name, state) {
      addStage(name, state);
      const dot = document.getElementById(`dot-${name}`);
      if (dot) { dot.className = `dot ${state}`; }
    }

    async function startJob(e) {
      e.preventDefault();
      submitBtn.disabled = true;
      statusEl.innerHTML = '';
      errorEl.textContent = '';
      resultEl.style.display = 'none';
      stageEls = {};

      const body = {
        topic: document.getElementById('topic').value,
        effort: document.getElementById('effort').value,
        audience: document.getElementById('audience').value,
        tone: document.getElementById('tone').value,
        theme: document.getElementById('theme').value,
        template: document.getElementById('template').value || null,
        speed: parseFloat(document.getElementById('speed').value) || 1.0,
      };

      const resp = await fetch('/api/jobs', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify(body),
      });

      if (!resp.ok) {
        errorEl.textContent = 'Failed to start job.';
        submitBtn.disabled = false;
        return;
      }

      const job = await resp.json();
      currentJobId = job.id;
      listenForEvents(job.id);
    }

    function listenForEvents(jobId) {
      const es = new EventSource(`/api/jobs/${jobId}/events`);

      es.onmessage = (e) => {
        const data = JSON.parse(e.data);

        if (data.done) {
          es.close();
          loadResult(jobId);
          submitBtn.disabled = false;
          return;
        }

        const node = data.node;
        if (!node || node === '__end__') return;

        const updates = data.updates || {};
        const status = updates.status || '';

        if (status === 'drafting' || status === 'validating') {
          setStage(node, 'running');
        } else {
          setStage(node, 'done');
        }
      };

      es.onerror = () => {
        es.close();
        loadResult(jobId);
        submitBtn.disabled = false;
      };
    }

    async function loadResult(jobId) {
      const resp = await fetch(`/api/jobs/${jobId}`);
      const job = await resp.json();

      if (job.status === 'failed') {
        errorEl.textContent = `Job failed: ${job.error || 'unknown error'}`;
        return;
      }

      if (job.status === 'completed' && job.output_files.includes('final.mp4')) {
        const videoEl = document.getElementById('video-player');
        videoEl.src = `/api/jobs/${jobId}/files/final.mp4`;
        resultEl.style.display = 'block';

        const dl = document.getElementById('downloads');
        dl.innerHTML = '';
        const downloadable = ['final.mp4', 'captions.srt', 'quiz.json', 'script.txt'];
        for (const f of downloadable) {
          if (job.output_files.includes(f)) {
            const a = document.createElement('a');
            a.href = `/api/jobs/${jobId}/files/${f}`;
            a.textContent = f;
            a.download = f;
            dl.appendChild(a);
          }
        }
      }
    }

    form.addEventListener('submit', startJob);
  </script>
</body>
